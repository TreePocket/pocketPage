    # day25-使用js实现选择排序

    使用JavaScript实现选择排序

    // 测试数据

    let arr = [5, 3, 2, 4, 1];

    console.log(selectionSort(arr)); // 输出: [1, 2, 3, 4, 5]

    function selectSort(arr){
  for(let i=0;i<arr.length;i++){
    let min=arr[i];
    let minIdx = i;
    for(let j=i;j<arr.length;j++){
      if(arr[j]<min){
        min = arr[j]
        minIdx = j
      }
    }
    if(minIdx !== i){
       [arr[i],arr[minIdx]] = [arr[minIdx],arr[i]]
    }
  }
}
